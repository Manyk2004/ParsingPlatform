# Установка
## Установка npm
- curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.7/install.sh | bash 
- cd .nvm/
## Установка node vite
- nvm install 20
- cd my_progect
- npm create vite@latest 
- npm install
## Установка дополнительных библиотек
- npm install antd --save
- npm install axios
